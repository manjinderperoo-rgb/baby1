import logging
from telegram import Update
from telegram.ext import (
    Application,
    CommandHandler,
    ContextTypes,
)

# ============ CONFIG ============
BOT_TOKEN = "7842373448:AAGejxy0Enq7Q27NfNk44KoFSPicd3vtBoU"
OWNER_ID = 5193826370          # apna telegram user id
FORCE_CHANNEL = "@techyspyther"

AUTHORIZED_GROUPS = set()

# ============ LOGGING ============
logging.basicConfig(level=logging.INFO)

# ============ FORCE JOIN CHECK ============
async def is_user_joined(bot, user_id: int) -> bool:
    try:
        member = await bot.get_chat_member(FORCE_CHANNEL, user_id)
        return member.status in ("member", "administrator", "creator")
    except:
        return False

async def force_join_guard(update: Update, context: ContextTypes.DEFAULT_TYPE):
    chat_id = update.effective_chat.id
    user_id = update.effective_user.id

    # group authorized hai ya nahi
    if chat_id not in AUTHORIZED_GROUPS:
        return False

    joined = await is_user_joined(context.bot, user_id)

    if not joined:
        await update.message.reply_text(
            f"‚ùå Pehle channel join karo:\n"
            f"https://t.me/{FORCE_CHANNEL.replace('@','')}"
        )
        return False

    return True

# ============ COMMANDS ============
async def add_group(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if update.effective_user.id != OWNER_ID:
        return

    chat_id = update.effective_chat.id
    AUTHORIZED_GROUPS.add(chat_id)

    await update.message.reply_text("‚úÖ Group authorized!")

async def test(update: Update, context: ContextTypes.DEFAULT_TYPE):
    allowed = await force_join_guard(update, context)
    if not allowed:
        return

    await update.message.reply_text("üî• Command executed successfully!")

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("ü§ñ Bot alive!")

# ============ MAIN ============
def main():
    app = Application.builder().token(BOT_TOKEN).build()

    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("add", add_group))
    app.add_handler(CommandHandler("test", test))

    print("ü§ñ Bot running...")
    app.run_polling()

if __name__ == "__main__":
    main()
